    src/Schema.cpp
    src/StorageManager.cpp
    src/Table.cpp
)

set(TEST_SOURCES
    tests/test_main.cpp
)

add_executable(tinydb ${SOURCES} main.cpp)

add_test(NAME tinydb_test COMMAND main_tests)
