# Add the include directory

file(GLOB SOURCES "src/*.cpp")

file(GLOB TEST_HPP "tests/*.hpp")


add_executable(main ${SOURCES} main.cpp)

# Register the test with CTest
add_test(NAME tinydb_test COMMAND t)
